import random
import time

def play_game():
    print("\n" + "="*40)
    print("     WELCOME TO GUESS THE NUMBER!")
    print("="*40)
    
    # Choose difficulty
    print("\nChoose difficulty level:")
    print("1 - Easy   (1–50, 12 attempts)")
    print("2 - Medium (1–100, 10 attempts)")
    print("3 - Hard   (1–500, 8 attempts)")
    
    while True:
        diff = input("Enter 1, 2 or 3: ").strip()
        if diff in ["1", "2", "3"]:
            diff = int(diff)
            break
        print("Please enter 1, 2 or 3 only.")

    if diff == 1:
        max_num = 50
        max_attempts = 12
        level = "Easy"
    elif diff == 2:
        max_num = 100
        max_attempts = 10
        level = "Medium"
    else:
        max_num = 500
        max_attempts = 8
        level = "Hard"

    secret = random.randint(1, max_num)
    attempts = 0
    start_time = time.time()

    print(f"\nI've picked a number between 1 and {max_num} ({level} mode).")
    print(f"You have {max_attempts} attempts. Good luck!\n")

    while attempts < max_attempts:
        attempts += 1
        remaining = max_attempts - attempts
        
        guess_input = input(f"Attempt {attempts}/{max_attempts} → Your guess: ").strip()
        
        if not guess_input.isdigit():
            print("→ Please enter a positive number!")
            attempts -= 1  # don't count invalid input as an attempt
            continue
            
        guess = int(guess_input)
        
        if guess == secret:
            time_taken = round(time.time() - start_time, 1)
            print("\n" + "★"*30)
            print(f"  YOU WIN! Congratulations!")
            print(f"  The number was {secret}")
            print(f"  Solved in {attempts} attempts ({time_taken} seconds)")
            print("★"*30)
            return attempts  # return attempts count for high-score tracking
        
        elif guess < secret:
            hint = "Too low ↑"
            emoji = "📈"
        else:
            hint = "Too high ↓"
            emoji = "📉"
        
        print(f"{emoji} {hint}")
        
        if remaining > 0:
            print(f"  {remaining} attempts left\n")
        else:
            break

    # Game over (loss)
    print("\n" + "×"*40)
    print(f"  GAME OVER! The secret number was {secret}")
    print("×"*40)
    return None  # loss → no high-score update


# Main game loop
best_score = float('inf')   # will be updated on wins
player_name = input("What's your name, champion? ").strip() or "Player"

print(f"\nHi {player_name}! Let's see how good you are 😎\n")

while True:
    score = play_game()
    
    if score is not None and score < best_score:
        best_score = score
        print(f"→ NEW RECORD! {best_score} attempts ←")
    
    if best_score != float('inf'):
        print(f"Current best: {best_score} attempts")
    
    again = input("\nPlay again? (y/n): ").lower().strip()
    if again not in ["y", "yes", ""]:
        print(f"\nThanks for playing, {player_name}!")
        print("See you next time! 👋")
        break
